\begin{proofp}
Let $s(n,k)$ denote the signless Stirling numbers of the first kind.  Using the recursive formulat $s(n,k)=s(n-1,k-1)+(n-1)s(n-1,k)$, prove that $$s(n,n-1)=\binom{n}{2}$$.
\end{proofp}

\begin{proofp}

\end{proofp}

\end{proofp}
